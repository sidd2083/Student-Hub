# ✅ FIX 1 — INSTANT EXISTING USER REDIRECT

Use onAuthStateChanged(auth, async (user) => {
   if (!user) return

   const docRef = doc(db, "users", user.uid)
   const snap = await getDoc(docRef)

   if (snap.exists()) {
      navigate("/dashboard")
   } else {
      navigate("/setup-profile")
   }
})
